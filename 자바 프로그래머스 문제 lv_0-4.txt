문제 61. 첫 번째로 나오는 음수
class Solution {
    public int solution(int[] num_list) {
        int answer = 0;
        for (int i =0; i<num_list.length;i++){
            if(num_list[i] < 0){
                return answer = i;
            }
            else {
                answer = -1;
            }
        }
        return answer;
    }
}

문제 62. 문자열의 뒤의 n글자
class Solution {
    public String solution(String my_string, int n) {
        String answer = "";
        answer = my_string.substring(my_string.length()-n);
        return answer;
    }
}
