    *** Swiper 메뉴슬라이드 (세로) ***
    const topHeaderSwiper = new Swiper('.top-header>.swiper', {
      direction: 'vertical', // 슬라이드가 위/아래로 움직이게 설정!
      autoplay: {
        delay: 2000,         // 2초마다 자동으로 다음 공지로 넘어가요.
      },
      loop: true,            // 마지막 공지 다음에 다시 첫 번째 공지가 나와요.
    });
    - 자동 전환 (autoplay): 사용자가 클릭하지 않아도 2초마다 새로운 공지가 올라옴
    - 방향 전환 (direction: 'vertical'): 기본값인 가로(horizontal)가 아닌 세로로 설정해서, 
                                         마치 전광판처럼 흐르는 효과
    - 무한 반복 (loop: true): 공지사항이 3번에서 끝나지 않고 다시 1번으로 자연스럽게 이어짐


    *** Swiper 멀티슬라이드 ***
    1. 멀티 슬라이더의 핵심 설정 (JavaScript)
    여러 개의 슬라이드를 동시에 보여주고 제어하기 위해 Swiper 인스턴스에 중요한 옵션들을 적용하셨어요.
    slidesPerView: 4:         한 화면에 4개의 슬라이드가 동시에 보이도록 설정했어요.
    centeredSlides: true:     현재 활성화된 슬라이드가 화면의 정중앙에 오도록 배치해서 시선을 집중시켜요.
    loop: true:               20개의 슬라이드가 끊기지 않고 무한히 회전하도록 만들었네요
        

    *** 02_LODASH ***
    window.addEventListener("scroll", _.throttle(()=>{console.log(window.scrollY);},100));
    = Lodash(로다쉬) 라이브러리를 사용해서 성능 최적화 코드(이벤트 체크 지연으로 성능 최적화)
    - Lodash에서 자주 쓰이는 대표 함수들
    함수명                설명                                    활용 예시
    _.throttle            일정 시간 간격으로 실행 제한             "스크롤 이벤트, 무한 스크롤 구현"
    _.debounce            마지막 이벤트 발생 후 일정 시간 대기      "검색창 입력(자동완성), 창 크기 조절"
    _.cloneDeep           객체의 깊은 복사 (참조값 분리)          복잡한 데이터 복제 및 수정
    _.uniq                배열 내 중복값 제거                     카테고리 목록 추출


    *** 03_SCROLLMAGIC ***
    new ScrollMagic
      .Scene({                                //감시 위치 옵션
        triggerElement: spyedEl,              //감시할 요소
        triggerHook: 0.5,                     //감시할 스크롤 위치값(0-1,0:최상단, 0.5:중앙, 1:최하단)
      })
      .setClassToggle(ballEl, 'move')         //애니메이션 효과 적용 위치
      .addTo(new ScrollMagic.Controller());   //제어부(스크롤 상황 감시)

    옵션 / 메서드	    역할	                    설정값
    triggerElement	    감시를 시작할 기준 요소	    .scroll-spy 섹션
    triggerHook	        뷰포트(화면) 내 트리거 위치	0.5 (화면 정중앙에 도달할 때)
    setClassToggle	    클래스를 넣었다 뺐다 함	    .ball 요소에 move 클래스 토글
    addTo	            컨트롤러에 장면을 등록	    new ScrollMagic.Controller()
