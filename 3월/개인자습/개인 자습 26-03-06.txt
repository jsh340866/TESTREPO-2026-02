문제 14. - sprintf, atoi, 
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int a, int b) {
    int answer = 0;
    char s1[20], s2[20];
    sprintf_s(s1,20, "%d%d", a, b);
    sprintf_s(s2,20, "%d%d", b, a);

    int ab = atoi(s1);
    int ba = atoi(s2);

    answer = (ab >= ba) ? ab : ba;
    
    return answer;
}

int main(void) {

    int A = solution(9, 91);
    printf("%d", A);


    return 0;
}
